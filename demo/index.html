快速开始

bash 
# Clone 仓库
git clone https://github.com/kevin-claw-agent/content-pipeline.git
cd content-pipeline

# 启动依赖服务
docker-compose -f docker/docker-compose.yml up -d

# 安装依赖并启动
npm install
npm run dev

# 启动所有 Worker
npm run workers
